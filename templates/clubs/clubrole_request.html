# models.py from django.db import models from django.contrib.auth.models import User class Club(models.Model): name = models.CharField(max_length=100) description = models.TextField() members = models.ManyToManyField(User, through='ClubMember') def __str__(self): return self.name class ClubMember(models.Model): user = models.ForeignKey(User, on_delete=models.CASCADE) club = models.ForeignKey(Club, on_delete=models.CASCADE) role = models.CharField(max_length=50) def __str__(self): return f"{self.user.username} - {self.club.name} ({self.role})" class RoleRequest(models.Model): ROLE_CHOICES = [ ('Club Admin', 'Club Admin'), ('Faculty Coordinator', 'Faculty Coordinator'), ] user = models.ForeignKey(User, on_delete=models.CASCADE) club = models.ForeignKey(Club, on_delete=models.CASCADE) role_requested = models.CharField(max_length=30, choices=ROLE_CHOICES) is_approved = models.BooleanField(default=False) timestamp = models.DateTimeField(auto_now_add=True) def __str__(self): return f"{self.user.username} requests {self.role_requested} for {self.club.name}" # views.py from django.shortcuts import render, get_object_or_404, redirect from django.contrib.auth.decorators import login_required from django.contrib.auth.models import Group from .models import Club, ClubMember, RoleRequest from django.http import HttpResponseRedirect from django.urls import reverse @login_required def club_members(request, club_id): club = get_object_or_404(Club, id=club_id) members = club.clubmember_set.all() user_groups = request.user.groups.values_list('name', flat=True) can_add_members = 'Club Admin' in user_groups or 'Faculty Coordinator' in user_groups role_requests = RoleRequest.objects.filter(club=club, is_approved=False) if 'Admin' in user_groups: visible_requests = role_requests elif 'Faculty Coordinator' in user_groups: visible_requests = role_requests.filter(role_requested='Club Admin') else: visible_requests = None return render(request, 'clubs/club_members.html', { 'club': club, 'members': members, 'can_add_members': can_add_members, 'visible_requests': visible_requests, }) @login_required def approve_role_request(request, request_id): req = get_object_or_404(RoleRequest, id=request_id) if request.method == "POST": group = Group.objects.get(name=req.role_requested) req.user.groups.add(group) req.is_approved = True req.save() return HttpResponseRedirect(reverse('club_members', args=[req.club.id])) @login_required def request_role(request, club_id): club = get_object_or_404(Club, id=club_id) if request.method == "POST": role = request.POST.get('role') existing = RoleRequest.objects.filter(user=request.user, club=club, role_requested=role, is_approved=False) if not existing.exists(): RoleRequest.objects.create(user=request.user, club=club, role_requested=role) return redirect('club_members', club_id=club.id) #
